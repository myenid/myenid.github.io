__int64 sub_400E00()
{
  char *v0; // rbx

  v0 = sub_400D74();                            // 合法字符%aAbBcCdDeFgGhHIjklmNnNpPrRsStTuUVwWxXyYzZ:-_/0^# 最后一个是空格
  if ( (unsigned int)sub_400CB5(v0) )           // 判读v0是不是全都在合法字符里面
  {
    ptr = v0;                                   // 字符串堆空间指针给全局变量
    puts("Format set.");
  }
  else
  {
    puts("Format contains invalid characters.");
    sub_400C7E(v0);
  }
  return 0LL;
}

char *sub_400D74()
{
  __int64 v0; // rdx
  __int64 v1; // rcx
  char s[1024]; // [rsp+8h] [rbp-410h]
  unsigned __int64 v4; // [rsp+408h] [rbp-10h]

  v4 = __readfsqword(0x28u);
  __printf_chk(1LL, "%s");
  fflush(stdout);
  fgets(s, 1024, stdin);
  s[strcspn(s, "\n")] = 0;                      // \n->\x00
  return sub_400C26(s, (__int64)"\n", v0, v1);
}

char *__fastcall sub_400C26(const char *a1, __int64 a2, __int64 a3, __int64 a4)
{
  char *v4; // rax
  char *v5; // rbx
  __int64 v7; // [rsp-8h] [rbp-18h]

  v7 = a4;                                      // 大小为malloc(strlen(s)+1)
  v4 = strdup(a1);                              // 分配一片堆空间来存储字符串
  if ( !v4 )
    err(1, "strdup", v7);
  v5 = v4;
  if ( getenv("DEBUG") )
    __fprintf_chk(stderr, 1LL, "strdup(%p) = %p\n", a1);
  return v5;
}

void __fastcall sub_400C7E(void *ptr)
{
  if ( getenv("DEBUG") )
    __fprintf_chk(stderr, 1LL, "free(%p)\n", ptr);
  free(ptr);
}

在show的里面，这样去输出时间。有个危险的system函数。然后我尝试了一下这个。

linux管道符号可以当前一个出错的时候执行后一个，那么我们可以控制ptr为 || /bin/sh

然后让时间报错，就可以成功执行了，继续看这个退出函数。

这两个free很妙，先free 时间格式，再free时间zone，还有个yes和no的判定，那么我们可以假装退出，然后选择n0,形成

ptr->value然后再去设置一次格式（报错），和zone,这样value->ptrzone指向的因该是和ptr同样的内存。

然后写入’||/bin/sh应该可以拿到shell

然后发现行不通，因为'没有闭合,导致系统会继续等待输入。

所以只能利用执行两句话的方法来了,加一个;执行两个指令。
