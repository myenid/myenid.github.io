    int j = 0;

    int flag = 0;

    while(j!=len)

        {
            if((arry[j] ^ arry[k]) == 0)
            {
                flag = 1;
                vims[k] = 1;
            }

        if(!flag && vims[j]!=1)

主元素

void searchmain()
{
    int n;
    cin >> n;
    int *A = new int[n];
    for(int i = 0; i < n; i++)
    {
        cin >> A[i];
    }
    int counts[10001];
    int vims[10001];
    for(int i = 0;i<n;i++)
    {
        for(int j = i; j<n;j++)
        {
            if((A[i] ^ A[j])== 0) counts[i]++;
        }
    }
    int flag = 0;
    int k = 0;
    while(k<n)
    {
        if(counts[k] > counts[flag]) flag = k;
        k++;
    }
    sort(counts,counts+n);
    if(counts[n-1]*2 > n) cout<<"主元素为"<< A[flag];
    else cout <<"没有主元素";
}

判断二叉排序树

这题有点难，，，首先给出的输入输出很鬼，然后不好去表示一个节点和树链接，不建树的话，直接看答案吧

当时我的想法是，给一个二维数组表示左右节点，然后再来一个数组表示当前节点的数据。但是一直没通过，应该是没有控制好递归的结束条件，和把握判断关键。

判断关键应该是左边最大值和右边最小值和当前值的一个比较，

给答案

#include<bits/stdc++.h> 
using namespace std;
struct node
{
	int l,r,data;
}t[1005];
bool vis[1005];
int mina[1005],maxa[1005];
int findlmax(int x)
{	if(t[x].l == -1) return t[x].data;	
	if(vis[x]) return maxa[x];
	return max(findlmax(t[x].l),t[x].data);
}
int findrmin(int x)
{	
	if(t[x].r == -1) return t[x].data;	
	if(vis[x]) return mina[x];
	return min(findrmin(t[x].r),t[x].data);
}		
int main()
{	
	int n;
	cin>>n;
	for(int i = 1;i<=n;++i)
	cin>>t[i].l>>t[i].r>>t[i].data;
	
	for(int i = 1;i<=n;++i)
	{
		if(!vis[i])
		{mina[i] =  findrmin(i);
		 maxa[i] =  findlmax(i);
		 vis[i]=1;
		 if(mina[i]<t[i].data||maxa[i]>t[i].data)
		 {
		 	cout<<"NO"<<endl;
		 	return 0;
		 }
		}
	}
	cout<<"YES"<<endl;
	fclose(stdin);
	fclose(stdout);
	return 0;
}
