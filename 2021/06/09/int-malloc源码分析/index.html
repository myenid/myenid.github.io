#define checked_request2size(req, sz) \
({				    \
  (sz) = request2size (req);	    \
  if (((sz) < (req))		    \
      || REQUEST_OUT_OF_RANGE (sz)) \
    {				    \
      __set_errno (ENOMEM);	    \
      return 0;			    \
    }				    \
})

#define request2size(req)                                         \
  (((req) + SIZE_SZ + MALLOC_ALIGN_MASK < MINSIZE)  ?             \
   MINSIZE :                                                      \
   ((req) + SIZE_SZ + MALLOC_ALIGN_MASK) & ~MALLOC_ALIGN_MASK)
//mask一看就是掩码0x111后面这个 与 符号应该就是控制对齐
//每次要够0x1000才多1此外，这里的SIZE_SZ + MALLOC_ALIGN_MASK
//也保证了chunk头
//例如req = 0x20 那么返回的就是0x30

#define MINSIZE  \
  (unsigned long)(((MIN_CHUNK_SIZE+MALLOC_ALIGN_MASK) & ~MALLOC_ALIGN_MASK))
//chunk头这里算出来是15

/* The smallest possible chunk */
#define MIN_CHUNK_SIZE        (offsetof(struct malloc_chunk, fd_nextsize))
//这是什么意思不知道，offset的写法没见过，但是含义是fd_nextsize的偏移16
//这里的意思是拿chunk头的大小的意思应该

/* The corresponding bit mask value.  */
#define MALLOC_ALIGN_MASK (MALLOC_ALIGNMENT - 1)

#define MALLOC_ALIGNMENT (2 * SIZE_SZ < __alignof__ (long double) \
			  ? __alignof__ (long double) : 2 * SIZE_SZ)
			 //这里的__alignof__是指对齐的意思，但是我推了一下，
			 //发现无论如何这都是16

具体的函数写的比较清楚，不一一展开这里只看alloc_perturb和tcache_put

static __always_inline void
tcache_put (mchunkptr chunk, size_t tc_idx)
{
  tcache_entry *e = (tcache_entry *) chunk2mem (chunk);
  //chun2mem是把指针+2*size_t
  assert (tc_idx < TCACHE_MAX_BINS);
  e->next = tcache->entries[tc_idx];
  tcache->entries[tc_idx] = e;
  ++(tcache->counts[tc_idx]);
}

可以看到放chunk进去的时候，除了idx没有任何检查，只要我们的idx符合就可以伪造任意的tcache进入bins。

static void
alloc_perturb (char *p, size_t n)
{
  //perturb_byte扰动字节，用来检测malloc和free尚未初始化的空间的错误
  //mmap或者brk获得堆空间之后，会填充perturb_byte^0xff而不是0
  if (__glibc_unlikely (perturb_byte))
    memset (p, perturb_byte ^ 0xff, n);
}

相应的有一个free_perturb

static void
free_perturb (char *p, size_t n)
{
  if (__glibc_unlikely (perturb_byte))
    memset (p, perturb_byte, n);
}

利用点

  • 首先是fastchunk的判断
size_t victim_idx = fastbin_index (chunksize (victim));
	      if (__builtin_expect (victim_idx != idx, 0))
		malloc_printerr ("malloc(): memory corruption (fast)");

#define SIZE_BITS (PREV_INUSE | IS_MMAPPED | NON_MAIN_ARENA)

/* Get size, ignoring use bits */
#define chunksize(p) (chunksize_nomask (p) & ~(SIZE_BITS))
/* Like chunksize, but do not mask SIZE_BITS.  */
#define chunksize_nomask(p)         ((p)->mchunk_size)

有个检查，所以伪造fastbin进去之后，malloc的时候会遇到这个检查，所以在伪造的时候需要idx能对上，再继续看比较的话，SIZE_BITS好像是没有影响的，在补码表示下~1 = 0x全f所以实际上chunksize()函数返回的就是chunk->msize这个东西，也就是我们要伪造的。所以伪造fastbin之后malloc的时候，一定要idx对的上此外任意地址malloc的时候要伪造chunk大小。

  • 对于tcache的利用

这里可以说是百无禁忌了，伪造没有任何检查，只需要保持idx<64就可以了，任意地址的malloc都可以。

  • 其次如果可以改tcache的max_idx，那么可以越界伪造
