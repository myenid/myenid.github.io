signed __int64 sub_4009E8()
{
  signed __int64 result; // rax
  int i; // eax
  unsigned int v2; // [rsp+8h] [rbp-88h]
  __int64 size; // [rsp+10h] [rbp-80h]
  char *ptr; // [rsp+18h] [rbp-78h]
  char s; // [rsp+20h] [rbp-70h]
  unsigned __int64 v6; // [rsp+88h] [rbp-8h]

  v6 = __readfsqword(0x28u);
  fgets(&s, 16, stdin);
  v2 = atol(&s);
  if ( v2 > 0x100000 )
    return 0xFFFFFFFFLL;
  if ( !Gloable[v2] )                           // v2是索引
    return 0xFFFFFFFFLL;                        // 判断索引位置是否存在内容
  fgets(&s, 16, stdin);
  size = atoll(&s);
  ptr = Gloable[v2];                            // 创建指向索引位置的指针
  for ( i = fread(ptr, 1uLL, size, stdin); i > 0; i = fread(ptr, 1uLL, size, stdin) )
  {
    ptr += i;
    size -= i;                                  // 循环输入，没有对size的长度做限制，导致溢出
  }
  if ( size )
    result = 0xFFFFFFFFLL;
  else
    result = 0LL;
  return result;
}

这段代码的难点就是 fread函数的返回值，怎么搜索也搜不到，最后还是读了一下源码，才弄清楚返回值。

附上源码

_IO_size_t
_IO_fread (void *buf, _IO_size_t size, _IO_size_t count, _IO_FILE *fp)
{
  _IO_size_t bytes_requested = size * count;
  _IO_size_t bytes_read;
  CHECK_FILE (fp, 0);
  if (bytes_requested == 0)
    return 0;
  _IO_acquire_lock (fp);
  bytes_read = _IO_sgetn (fp, (char *) buf, bytes_requested);
  _IO_release_lock (fp);
  return bytes_requested == bytes_read ? count : bytes_read / size;
}

这里不去深入剖析他的作用了。fread函数有四个参数。buf是指针，size是数据的大小，count表示要传入多少个对象（数据），fp表示输入输出流对象。作用就是从指定的流里面传入count个size大小的对象给buf。

返回值我们看一下，传入失败当然市return 0；

当传入成功的时候return bytes_requested == bytes_read ? count : bytes_read / size;其中bytes_read = size *count，所以返回的是count。

那么我们再次返回fill代码去分析，每次循环都会有个ptr += i这里的 i 就是 size 那么每次我们循环后，就发现size变为0了，所以虽然是循环但每次都只输入一次。

分析

思路：

  • RELRO为Partial RELRO，那么可以改写got表；
  • 没有system等后门函数，所以得想办法先泄漏libc，然后调用libc里的system函数；
  • 要泄漏？那么我们看看打印和输入的地方有哪些：
    • 打印：show函数没有打印堆里的内容，没法leak，所以我们得想办法leak。
    • 输入：有一个fill函数，可以往bss段里globals数组中那些指针所指向的地方写。
  • 而fill函数中存在堆溢出，可以利用unlink漏洞修改globals数组！程序里有puts函数，所以我们可以把puts函数plt地址放到另一个函数的got表项里，这样，调用这个函数的时候就会去调用puts函数。至于这个函数选哪个，因为puts函数参数是char型指针，所以我们可以选取程序里有的，且参数同样为指针的free/atoi函数。但是如果用atoi函数，那么我们后面就没法用free函数了，因为输入“菜单选项”后就会调用atoi函数，没法继续下去。所以，我们选free函数。
  • 下面两图是把puts@plt放入free函数的got表项里。那么这时候调用free(globals[2])的话，就等价于puts(puts@got)，会调用puts函数把puts@got的内容打印出来。
  • 得到puts函数真实地址后，就可以计算libc基址及system函数地址了，最后把system地址通过fill函数写入atoi@got，把“/bin/sh”当作“菜单选项”输入，就会调用atoi函数，也就是调用system(“/bin/sh”)以getshell！

这里有个疑惑，为什么修改的是chunk而不是数组想了巨久，然后悟了。

这是个指针数组，两个指针传值的话改变的是指向的内容。

//举个例子
ptr1 和 ptr2是指针
ptr1指向 0xa
ptr2指向 0xb
ptr1=ptr2实现的是ptr也指向0xb。

所以函数内,ptr = gloabal[v2]传递的是指向的内容。所以这时候ptr指向了chunk，所以总的来说，修改的还是chunk。

这里就利用到了原来how2heap上学的unlink的伪造利用了。gloabal就是那个全局指针。才有了上面的构造

构造过程总结一下

  • 首先，我确定要修改哪里的内存，就是要在哪里去修改一个变量
  • 这里没有选择第一个是因为，该程序没有setbug，导致一开始会先给gets之类的函数分配内存，unlink可能不是特别好用，所以选择中间的堆
  • 然后，这个我们确定要修改的就是，我们伪造chunk的前一个假chunk（我理解是伪造了两个chunk）。所以很容易得出，ptr_fd和ptr_bk的位置。 始终记住我们要改的位置，是最高地址，那么伪造一个chunk在chunk2里面的话，伪造的fd, fake_fd->bk就是我们在bss段的虚假头指针，要满足fake_fd->bk->fd = fake_chunk的话，我们就需要让fake_fd->bk->fd刚好是aim(我们要修改位置)，因为aim指向mem也就是我们的fake_chunk.
  • 这时候，其实我们能修改的位置是固定的。
  • 回到第二点，而fake_fd->bk的fd位置（fake_fd->bk指向的是一个chunk）是 &fake_fd->bk + 0x10,所以aim-0x10的位置必须是fake_fd指向的bk，所以我们就覆盖fake_fd的值为aim-0x10。同理可以计算fd的位置为aim-0x18

最后记得修改下一个chunk的pre_size位和size位的最低位。让系统以为我们的fake_chunk是free状态的。

接着free(chunk3)

free(bss+0x18)

free之后就有一些神奇的事情发生了。会对fake_chunk进行unlink.

也就是

fake_chunk_fd->bk = bk; fake_bk->fd = fd这样操作之后，aim里面存着的就是aim-0x18了，意思是aim里面存折aim-0x18的地址，aim指向aim-0x18。

然后我们再次edit 2号堆，实际上就是对bss段的内存进行任意的读写了。

然后来了更加妙的操作。

这样去构造堆块，然后edit 0，可以改free的got表，然后free(1)就泄露出了puts函数的got表地址，再edit 2修改atoi的got表为system函数的地址，然后输入/bin/sh。atoi(“/bin/sh”)就相当于 system(“/bin/sh”)，就拿到了shell。
