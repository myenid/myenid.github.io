unsigned __int64 Create_chunk()
{
  signed int i; // [rsp+4h] [rbp-102Ch]
  char *ptr; // [rsp+8h] [rbp-1028h]
  char *dest; // [rsp+10h] [rbp-1020h]
  size_t nbytes; // [rsp+18h] [rbp-1018h]
  size_t nbytesa; // [rsp+18h] [rbp-1018h]
  char buf; // [rsp+20h] [rbp-1010h]
  unsigned __int64 v7; // [rsp+1028h] [rbp-8h]

  v7 = __readfsqword(0x28u);
  ptr = malloc(0x20uLL);
  printf("Pls give string size:");
  nbytes = get_num();
  if ( nbytes <= 0x1000 )
  {
    printf("str:");
    if ( read(0, &buf, nbytes) == -1 )
    {
      puts("got elf!!");
      exit(1);
    }
    nbytesa = strlen(&buf);
    if ( nbytesa > 0xF )
    {
      dest = malloc(nbytesa);                   // 申请chunk
      if ( !dest )
      {
        puts("malloc faild!");
        exit(1);
      }
      strncpy(dest, &buf, nbytesa);
      *ptr = dest;
      *(ptr + 3) = Free_addr_2;
    }
    else
    {
      strncpy(ptr, &buf, nbytesa);
      *(ptr + 3) = Free_addr;
    }
    *(ptr + 4) = nbytesa;                       // 这里应该是+2
    for ( i = 0; i <= 15; ++i )
    {
      if ( !*(&note_list + 4 * i) )             // 给全局变量赋值吧，调试才知道结果
      {
        *(&note_list + 4 * i) = 1;
        *(&note_list + 2 * i + 1) = ptr;
        printf("The string id is %d\n", i);
        break;
      }
    }
    if ( i == 16 )
    {
      puts("The string list is full");
      (*(ptr + 3))(ptr);
    }
  }
  else
  {
    puts("Invalid size");
    free(ptr);
  }
  return __readfsqword(0x28u) ^ v7;
}

unsigned __int64 del_chunk()
{
  int v1; // [rsp+Ch] [rbp-114h]
  char buf; // [rsp+10h] [rbp-110h]
  unsigned __int64 v3; // [rsp+118h] [rbp-8h]

  v3 = __readfsqword(0x28u);
  printf("Pls give me the string id you want to delete\nid:");
  v1 = get_num();
  if ( v1 < 0 || v1 > 16 )
    puts("Invalid id");
  if ( *(&note_list + 2 * v1 + 1) )
  {
    printf("Are you sure?:");
    read(0, &buf, 0x100uLL);
    if ( !strncmp(&buf, "yes", 3uLL) )
    {
      (*(*(&note_list + 2 * v1 + 1) + 0x18LL))(*(&note_list + 2 * v1 + 1), "yes");
      *(&note_list + 4 * v1) = 0;               // 序号置为0，但是没有把指针置为0，导致可以UAF
    }
  }
  return __readfsqword(0x28u) ^ v3;
}

分析

先申请一个0x8大小的，这时候就会分配一个不是fastbin的chunk,标记为note1，申请一个一样的note2，然后free2,free1，之后申请一个0x20的note3，这时候，note1的free_addr位就会被覆盖。

泄露一次地址，然后再改成sys，就可以拿到shell了。
